Quick Start

Get Running in Minutes

Automated setup for Windows, macOS, and Linux

# Clone repository
git clone https://github.com/amuzetnoM/vector_studio.git
cd vector_studio

# Build
cmake -B build -DCMAKE_BUILD_TYPE=Release
cmake --build build

# Python usage
import pyvdb

# Create database
db = pyvdb.create_database("./vectors")

# Add text with metadata
db.add_text("High-performance vector search", {
    "type": "document",
    "date": "2025-01-04"
})

# Search
results = db.search("fast similarity search", k=10)
for r in results:
    print(f"{r.score:.4f}: {r.text}")
